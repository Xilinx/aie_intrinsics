acc1 = -( z10*(x10 + y10) + z11*(x11 + y11) + z12*(x12 + y12) + z13*(x13 + y13) )

acc4 = -( z40*(x40 + y40) + z41*(x41 + y41) + z42*(x42 + y42) + z43*(x43 + y43) )
